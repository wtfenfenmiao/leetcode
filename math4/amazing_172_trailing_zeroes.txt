//思路这样的，算有多少个0就是算有多少个5，然后n/5，是算有几个乘一个5的，就是5，10，15，20，25，30......首先都是乘1个5的（5*（1，2，3，4，5，6......））；
//然后在这里有25，50，75这种乘了两个5的(5*5*(1,2,3,4,5,6......))，就再n/5找有几个这种的；然后再算乘三个5的......
class Solution {
public:
    int trailingZeroes(int n) {
        int re=0;
        while(n>0)
        {
            n/=5;
            re+=n;
        }
        return re;
    }
};
