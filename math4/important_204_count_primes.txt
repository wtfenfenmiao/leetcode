class Solution {
public:
    int countPrimes(int n) {
        int count=0;
        vector<bool> list(n,false);
        for(int i=2;i<n;++i)
        {
            if(!list[i])
            {
                ++count;
                for(int j=2;i*j<n;++j)
                {
                    list[i*j]=true;
                }
            }
        }
        return count;
    }
};
