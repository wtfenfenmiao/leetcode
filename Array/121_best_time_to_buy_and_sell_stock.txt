//看代码就知道思路了，one pass
class Solution {
public:
    int maxProfit(vector<int>& prices) {
        if(prices.size()==0)
            return 0;
        int minprice=prices[0];
        int maxprofit=0;
        for(int i=0;i<prices.size();++i)
        {
            if(prices[i]<minprice)
                minprice=prices[i];
            else
                maxprofit=max(maxprofit,prices[i]-minprice);
        }
        return maxprofit;
    }
};
