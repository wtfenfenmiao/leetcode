/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    bool hasCycle(ListNode *head) {
        ListNode *slow=head;
        ListNode *fast=head;
        if(fast==NULL||fast->next==NULL)
            return false;
        do
        {
            slow=slow->next;
            fast=fast->next;
            fast=fast->next;          
        }while((fast!=NULL&&fast->next!=NULL)&&(fast!=slow));
        if(fast==slow)
            return true;
        else
            return false;
    }
};
