class Solution {
public:
    bool containsNearbyDuplicate(vector<int>& nums, int k) {
        unordered_map<int,int> hashMap;
        unordered_map<int,int>::iterator iter;
        for(int i=0;i<nums.size();++i)
        {
            iter=hashMap.find(nums[i]);
            if(iter!=hashMap.end()&&(i-iter->second)<=k)
                return true;
            else
            {
                hashMap[nums[i]]=i;
            }
        }
        return false;
    }
};
