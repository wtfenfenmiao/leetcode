class Solution {
public:
    bool isValid(string s) {
        stack<char> test;
        for(int i=0;i<s.size();++i)
        {
            if(s[i]=='(')
                test.push(')');
            else if(s[i]=='[')
                test.push(']');
            else if(s[i]=='{')
                test.push('}');
            else if(test.empty()||test.top()!=s[i])
            {
                return false;
            }
            else
                test.pop();               
        }
        return test.empty()?true:false;
    }
};

//同理又写的
class Solution {
public:
    bool isValid(string s) {
        stack<char> st;
        unordered_map<char,char> list={{')','('},{']','['},{'}','{'}};
        for(int i=0;i<s.size();++i)
        {
            if(s[i]=='('||s[i]=='{'||s[i]=='[')
                st.push(s[i]);
            else
            {
                if(st.empty()||st.top()!=list[s[i]])
                    return false;
                else
                    st.pop();
            }
        }
        return st.empty();
    }
};
