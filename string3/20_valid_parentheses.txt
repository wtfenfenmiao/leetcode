class Solution {
public:
    bool isValid(string s) {
        stack<char> test;
        for(int i=0;i<s.size();++i)
        {
            if(s[i]=='(')
                test.push(')');
            else if(s[i]=='[')
                test.push(']');
            else if(s[i]=='{')
                test.push('}');
            else if(test.empty()||test.top()!=s[i])
            {
                return false;
            }
            else
                test.pop();               
        }
        return test.empty()?true:false;
    }
};
