//从中间往两边导......
class Solution {
public:
    void sortColors(vector<int>& nums) {
        int zero=0;
        int second=nums.size()-1;
        for(int i=0;i<=second;++i)
        {
            while(nums[i]==2&&i<second)
                swap(nums,i,second--);
            while(nums[i]==0&&i>zero)
                swap(nums,i,zero++);
        }
    }
    void swap(vector<int>& nums,int i,int j)
    {
        int temp=nums[i];
        nums[i]=nums[j];
        nums[j]=temp;
    }
};
