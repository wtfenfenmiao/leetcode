class Solution {
public:
    vector<vector<int>> subsets(vector<int>& nums) {
        int sum=1 << nums.size();
        vector<vector<int>> re(sum,vector<int>());
        for(int i=0;i<nums.size();++i)
        {
            int k=0;
            for(int j=(1 << i);j<(1 << (i+1));++j)
            {
                re[j]=re[k++];     
                re[j].push_back(nums[i]);
            }
        }
        return re;
    }
};
